cmake_minimum_required(VERSION 3.12)
project(ChangeMyName CXX)

set(CMAKE_CXX_STANDARD 17)
add_compile_options(-Wall -Wextra -pedantic -Werror)

include(FetchContent)
FetchContent_Declare(
        googletest
        GIT_REPOSITORY https://github.com/google/googletest.git
        GIT_TAG main
)

FetchContent_MakeAvailable(googletest)

add_executable(main project/src/Main.cpp project/src/Board.cpp project/src/Player.cpp project/src/Game.cpp)
add_executable(sample_test project/src/PajonkTests.cpp project/src/Board.cpp project/src/Player.cpp project/src/Game.cpp)

target_include_directories(main PUBLIC include)

target_include_directories(sample_test PUBLIC include)
target_link_libraries(sample_test gmock_main gtest gmock)

add_custom_target(run_tests
                  COMMAND sample_test
                  DEPENDS sample_test)

enable_testing()
add_test(tests sample_test)
